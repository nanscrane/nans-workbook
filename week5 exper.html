Video drawing

var capture;
function setup() {
createCanvas(windowWidth, windowHeight);
imageMode(CENTER);
capture=createCapture(VIDEO);
capture.size(640,480);
//capture.hide();
imageMode(CENTER);
}
function draw() {
image(capture,mouseX,mouseY);
if (!mouseIsPressed) {
filter(THRESHOLD);
}
}
function mousePressed(){
image(capture,mouseX,mouseY);
filter(INVERT);
}
